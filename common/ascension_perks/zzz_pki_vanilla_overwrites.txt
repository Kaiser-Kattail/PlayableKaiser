    ai_weight = {
        factor = 7.5

                is_militarist = yes

                is_gestalt = yes
            }
        }
        modifier = {
            factor = 10
            is_homicidal = yes
        }
        modifier = {
            factor = 2
            has_tradition = tr_supremacy_finish
        }
        modifier = {
            factor = 0
            galaxy_percentage < 0.1
        }
        modifier = {
            factor = 2
            galaxy_percentage > 0.2
        }
        modifier = {
            factor = 0
            count_playable_country = {

                }
            }

        }
