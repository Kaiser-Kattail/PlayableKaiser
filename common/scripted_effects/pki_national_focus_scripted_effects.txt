pki_clear_gui_flags = {
    remove_country_flag = pki_conquest_map_tab
    remove_country_flag = pki_focus_tree_tab
}

pki_clear_focus_tree_tab_flags = {
    remove_country_flag = pki_focus_tree_tab_state
    remove_country_flag = pki_focus_tree_tab_military
    remove_country_flag = pki_focus_tree_tab_innovation
    remove_country_flag = pki_focus_tree_tab_galactic
}

inline_script = {
    script = pki_focuses/pki_focuses
    PART = effect
}

pki_focus_in_progress_effects = {
    switch = {
        trigger = has_situation_flag
        inline_script = {
            script = pki_focuses/pki_focuses
            PART = in_progress
        }
    }
}

pki_add_factories = {
    tooltip = {
        add_deposit = d_pki_factory_add_$VALUE$
    }
    hidden_effect = {
        set_variable = {
            which = pki_industry_level
            value = $VALUE$
        }
    }
}

pki_instant_focus = {
    if = {
        limit = {
            has_country_flag = pki_instant_focus
        }
        remove_country_flag = pki_instant_focus
    }
    else = {
        set_country_flag = pki_instant_focus
    }
}

pki_complete_focus = {
    every_situation = {
        limit = {
            is_situation_type = situation_naitonal_focus
        }
        add_situation_progress = 12
    }
}
