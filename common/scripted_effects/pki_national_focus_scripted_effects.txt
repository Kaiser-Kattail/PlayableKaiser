pki_clear_gui_flags = {
    remove_country_flag = pki_conquest_map_tab
    remove_country_flag = pki_focus_tree_tab
}

pki_complete_national_focus = {
    switch = {
        trigger = has_situation_flag
        pki_kaiser_arrival = {
           owner = {
               country_event = {
                    id = pki_focus.1
               }
               set_country_flag = pki_focus_kaiser_arrival_completed
           }
        }
        pki_neighbors = {
            owner = {
                set_country_flag = pki_focus_neighbors_completed
                add_modifier = {
                    modifier = pki_neighbors_modifier
                    years = 5
                }
            }
        }
        pki_harania = {
            owner = {
                set_country_flag = pki_focus_harania_completed
                declare_war = {
                    attacker_war_goal = wg_player_katzen
                    target = event_target:haranian_flunat
                }
            }
        }
    }
}
