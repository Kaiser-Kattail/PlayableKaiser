pki_is_katzenartig_imperium = {
	AND = {
		has_origin = origin_katzen
		has_country_flag = pki_playable_kaiser
	}
}

pki_is_katzprotektorat = {
	OR = {
		has_country_flag = pki_katzprotectorat
		has_country_flag = pki_katzprotectorat_gestalt
	}
}

pki_is_katzprotektorat_tier_2 = {
	OR = {
		is_specialist_subject_tier = { TYPE = katzprotektorat TIER = 2}
		is_specialist_subject_tier = { TYPE = katzprotektorat_gestalt TIER = 2}
	}
}

pki_is_katzprotektorat_tier_3 = {
	OR = {
		is_specialist_subject_tier = { TYPE = katzprotektorat TIER = 3}
		is_specialist_subject_tier = { TYPE = katzprotektorat_gestalt TIER = 3}
	}
}

pki_is_katzen_ship = {
	OR = {
		is_ship_size = pki_panzer
		is_ship_size = pki_bomber
		is_ship_size = pki_cruiser
	}
}

is_pki_flusion_ecu = {
	is_planet_class = pc_pki_flusion_ecu
}

is_pki_katzen_trait_elite = {
	has_trait = pki_katzen_trait_elite
}

pki_is_pki_army = {
	OR = {
		pki_is_pki_ground_army = yes
	}
}

pki_is_pki_ground_army = {
	OR = {
		army_type = pki_katzen_special_forces
		army_type = pki_katzen_tanks
		army_type = peacekeeper_army
		army_type = dominion_tracks
		army_type = meowrine_company
	}
}

pki_is_normal_army = {
	OR = {
		pki_is_pki_ground_army = yes

		### VANILLA
		army_type = assault_army
		army_type = slave_army
		army_type = clone_army
		army_type = robotic_army
		army_type = psionic_army
		army_type = xenomorph_army
		army_type = gene_warrior_army
		army_type = undead_army
		army_type = warpling_army
		army_type = flamestorm_troopers_army
	}
}

pki_is_air_wing = {
	OR = {
		army_type = pki_air_fighter
		army_type = pki_air_cas
	}
}

pki_is_allied_with_katzen = {
	any_federation_ally = {
		pki_is_katzenartig_imperium = yes
	}
}

pki_is_allowed_for_kaiser_spawn = {
	NOR = {
		AND = {
			exists = space_owner
			space_owner = { is_fallen_empire = yes }
		}
		any_neighbor_system = {
			OR = {
				# is_fe_cluster = yes
				has_star_flag = empire_home_system
				AND = {
					exists = space_owner
					space_owner = { is_fallen_empire = yes }
				}
				any_neighbor_system = {
					OR = {
						# is_fe_cluster = yes
						has_star_flag = empire_home_system
						AND = {
							exists = space_owner
							space_owner = { is_fallen_empire = yes }
						}
					}
				}
			}
		}
	}
}

can_destroy_planet_with_PKI_PLANET_KILLER_CRACKER = {
	can_destroy_planet_with_PLANET_KILLER_CRACKER = yes
}

pki_has_leader_cap_trait = {
	OR = {
		has_trait = pki_katzen_trait_elite

	}
